450  case OB_FRAME_COLOR:
451  return (typeid(T) == typeid(ColorFrame) || typeid(T) == typeid(VideoFrame));
452  case OB_FRAME_GYRO:
453  return (typeid(T) == typeid(GyroFrame));
454  case OB_FRAME_ACCEL:
455  return (typeid(T) == typeid(AccelFrame));
456  case OB_FRAME_SET:

458  case OB_FRAME_POINTS:
459  return (typeid(T) == typeid(PointsFrame));
